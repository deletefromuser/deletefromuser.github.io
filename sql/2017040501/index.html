sqlserver 修改主键约束

sp_help book

if exists (select 1 from sysobjects where name = 'PK__book__AFEBF550AD55A29F')
begin
	print 'pk exists'
	alter table book drop constraint PK__book__AFEBF550AD55A29F
end

alter table book add constraint PK__book primary key clustered
(
book_name asc,
id asc
)
